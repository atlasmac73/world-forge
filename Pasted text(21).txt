function SectionTitle({ children }) {

    <div style={{ fontSize: 11, color: "#00f5c4", letterSpacing: 4, marginBottom: 20, fontWeight: 700 }}>

      ◈ {children}

    </div>

  );

function PhaseTag({ status }) {

  const cfg = { active: ["#00f5c4", "ACTIVE"], planned: ["#facc15", "PLANNED"], future: ["#888", "FUTURE"] };

  const [color, label] = cfg[status] || ["#888", status];

  return (

    <span style={{

      fontSize: 9, padding: "2px 7px", borderRadius: 10,

      background: `${color}20`, color, letterSpacing: 1,

    }}>{label}</span>

  );

}



function PriorityBadge({ p }) {

  const cfg = { critical: "#ff4444", high: "#ff6b35", medium: "#facc15", low: "#888" };

  return (

    <span style={{

      fontSize: 9, padding: "1px 6px", borderRadius: 3,

      background: `${cfg[p] || "#888"}25`, color: cfg[p] || "#888",

      letterSpacing: 1,

    }}>{p?.toUpperCase()}</span>

  );

}
